```python
def test_build_llm_kwargs_minimal():
    """With no VLLMConfig, kwargs has model, dtype, trust_remote_code, seed."""
    config = _make_config()
    backend = VLLMBackend()
    kwargs = backend._build_llm_kwargs(config)
    assert kwargs["model"] == "test-model"
    assert kwargs["trust_remote_code"] is True
    assert kwargs["seed"] == 42
    assert "dtype" in kwargs

def test_build_llm_kwargs_with_vllm_config():
    """VLLMConfig fields are applied when not None."""
    vllm_cfg = VLLMConfig(
        tensor_parallel_size=2,
        gpu_memory_utilization=0.85,
        max_num_seqs=128,
        enable_prefix_caching=True,
        quantization="awq",
    )
    config = _make_config(vllm=vllm_cfg)
    backend = VLLMBackend()
    kwargs = backend._build_llm_kwargs(config)
    assert kwargs["tensor_parallel_size"] == 2
    assert kwargs["gpu_memory_utilization"] == 0.85
    assert kwargs["max_num_seqs"] == 128
    assert kwargs["enable_prefix_caching"] is True
    assert kwargs["quantization"] == "awq"

def test_build_llm_kwargs_none_fields_omitted():
    """VLLMConfig fields that are None are NOT included in kwargs."""
    vllm_cfg = VLLMConfig(tensor_parallel_size=2)  # only TP set
    config = _make_config(vllm=vllm_cfg)
    backend = VLLMBackend()
    kwargs = backend._build_llm_kwargs(config)
    assert kwargs["tensor_parallel_size"] == 2
    assert "gpu_memory_utilization" not in kwargs
    assert "max_num_seqs" not in kwargs

def test_build_llm_kwargs_precision_mapping():
    """precision field maps through _map_precision to dtype kwarg."""
    for precision, expected in [("fp32", "float32"), ("fp16", "float16"), ("bf16", "bfloat16")]:
        config = _make_config(precision=precision)
        kwargs = VLLMBackend()._build_llm_kwargs(config)
        assert kwargs["dtype"] == expected
```
